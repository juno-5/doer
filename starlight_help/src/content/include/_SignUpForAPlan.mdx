import {Steps, TabItem, Tabs} from "@astrojs/starlight/components";

If you require features that are not available on [Doer Cloud
Free](https://zulip.com/plans/#cloud) or the [Doer
Free](https://zulip.com/plans/#self-hosted) plan for self-hosted organizations,
you will need to upgrade your plan.

<Tabs>
  <TabItem label="Doer Cloud">
    <Steps>
      1. Follow the
         [instructions](/help/doer-cloud-billing#upgrade-to-a-doer-cloud-standard-or-plus-plan)
         to upgrade to a Doer Cloud Standard or Plus plan. If your organization may
         be
         [eligible](/help/doer-cloud-billing#free-and-discounted-doer-cloud-standard)
         for a free or discounted plan, you can [apply for
         sponsorship](/help/doer-cloud-billing#apply-for-sponsorship).
    </Steps>
  </TabItem>

  <TabItem label="Self hosting">
    <Steps>
      1. Follow the [instructions](/help/self-hosted-billing#upgrade-to-a-paid-plan)
         to upgrade to a Doer Basic or Doer Business plan. If your organization may
         be [eligible](/help/self-hosted-billing#free-community-plan) for a free or
         discounted plan, you can [apply for
         sponsorship](/help/self-hosted-billing#apply-for-community-plan). To inquire
         about Doer Enterprise, please reach out to
         [sales@zulip.com](mailto:sales@zulip.com).
    </Steps>
  </TabItem>
</Tabs>
